Getting Started

Implement a ValkeySessionRepository that plugs into Strands' official SessionManager interface — giving your agent persistent conversation history, session metadata, and agent state backed by Valkey.

How Strands Manages State

Strands has a formal session management interface. The key classes are:

  • SessionRepository — abstract base class defining the storage contract
  • RepositorySessionManager — the built-in session manager that delegates to any SessionRepository
  • Agent(session_manager=...) — pass a session manager and Strands handles the rest

Built-in backends are FileSessionManager and S3SessionManager. Valkey isn't bundled yet — but the SessionRepository ABC makes it straightforward to implement.

What Gets Stored

The session manager persists three types of objects to Valkey:

ObjectTypeContains
SessionSessionSession ID, creation time, metadata
Agent stateSessionAgentAgent ID, conversation manager state, interrupt state
MessagesSessionMessageRole, content, tool use/results, timestamps

Prerequisites

  • Docker installed
  • Python 3.10+
  • AWS credentials configured (Strands uses Amazon Bedrock in us-west-2 by default)

Step 3: Implement ValkeySessionRepository

Subclass SessionRepository and implement its abstract methods. Each method maps to a Valkey operation:

import json
import uuid
from datetime import datetime, timezone
import valkey

from strands.session.session_manager import RepositorySessionManager
from strands.session.repository import SessionRepository
from strands.types.session import Session, SessionAgent, SessionMessage


class ValkeySessionRepository(SessionRepository):
    """Valkey-backed implementation of the Strands SessionRepository interface."""

    def __init__(self, host: str = "localhost", port: int = 6379, ttl: int = 3600):
        self.client = valkey.Valkey(host=host, port=port, decode_responses=True)
        self.ttl = ttl  # seconds until session data expires

    # ── Key helpers ──────────────────────────────────────────────────────────
    def _session_key(self, session_id: str) -> str:
        return f"strands:session:{session_id}"

    def _agent_key(self, session_id: str, agent_id: str) -> str:
        return f"strands:session:{session_id}:agent:{agent_id}"

    def _message_key(self, session_id: str, message_id: str) -> str:
        return f"strands:session:{session_id}:msg:{message_id}"

    def _messages_list_key(self, session_id: str, agent_id: str) -> str:
        return f"strands:session:{session_id}:agent:{agent_id}:messages"

    # ── Session CRUD ─────────────────────────────────────────────────────────
    def create_session(self, session: Session) -> Session:
        key = self._session_key(session.session_id)
        self.client.set(key, json.dumps(session.model_dump()), ex=self.ttl)
        return session

    def read_session(self, session_id: str) -> Session | None:
        raw = self.client.get(self._session_key(session_id))
        if raw is None:
            return None
        return Session(**json.loads(raw))

    # ── Agent CRUD ───────────────────────────────────────────────────────────
    def create_agent(self, session_id: str, agent: SessionAgent) -> SessionAgent:
        key = self._agent_key(session_id, agent.agent_id)
        self.client.set(key, json.dumps(agent.model_dump()), ex=self.ttl)
        return agent

    def read_agent(self, session_id: str, agent_id: str) -> SessionAgent | None:
        raw = self.client.get(self._agent_key(session_id, agent_id))
        if raw is None:
            return None
        return SessionAgent(**json.loads(raw))

    def update_agent(self, session_id: str, agent: SessionAgent) -> SessionAgent:
        return self.create_agent(session_id, agent)  # SET overwrites

    # ── Message CRUD ─────────────────────────────────────────────────────────
    def create_message(self, session_id: str, agent_id: str, message: SessionMessage) -> SessionMessage:
        # Store message payload
        msg_key = self._message_key(session_id, message.message_id)
        self.client.set(msg_key, json.dumps(message.model_dump()), ex=self.ttl)
        # Append message ID to the ordered list for this agent
        list_key = self._messages_list_key(session_id, agent_id)
        self.client.rpush(list_key, message.message_id)
        self.client.expire(list_key, self.ttl)
        return message

    def read_message(self, session_id: str, agent_id: str, message_id: str) -> SessionMessage | None:
        raw = self.client.get(self._message_key(session_id, message_id))
        if raw is None:
            return None
        return SessionMessage(**json.loads(raw))

    def update_message(self, session_id: str, agent_id: str, message: SessionMessage) -> SessionMessage:
        return self.create_message(session_id, agent_id, message)

    def list_messages(self, session_id: str, agent_id: str) -> list[SessionMessage]:
        list_key = self._messages_list_key(session_id, agent_id)
        message_ids = self.client.lrange(list_key, 0, -1)
        messages = []
        for mid in message_ids:
            msg = self.read_message(session_id, agent_id, mid)
            if msg:
                messages.append(msg)
        return messages

Design note: Messages are stored as individual Valkey strings (one key per message) with an ordered list of IDs per agent. This lets you read a single message by ID in O(1) while still replaying the full conversation in order via LRANGE.

Step 4: Wire It to a Strands Agent

from strands import Agent

# Create the repository and session manager
repo = ValkeySessionRepository(host="localhost", port=6379, ttl=3600)
session_mgr = RepositorySessionManager(
    session_id="user-42",
    session_repository=repo,
)

# Pass the session manager to the agent
agent = Agent(
    system_prompt="You are a helpful assistant.",
    session_manager=session_mgr,
)

# Strands automatically persists every turn to Valkey
response = agent("My name is Alex and I'm building a RAG pipeline.")
print(response)

Step 5: Resume a Session

Create a new Agent with the same session_id — Strands reloads the full conversation history from Valkey automatically:

# In a new Python process — history is restored from Valkey
repo = ValkeySessionRepository(host="localhost", port=6379)
session_mgr = RepositorySessionManager(
    session_id="user-42",  # same session ID
    session_repository=repo,
)
agent = Agent(session_manager=session_mgr)

response = agent("What was I just telling you about?")
print(response)
# Agent correctly recalls: "You mentioned you're building a RAG pipeline."

Step 6: Inspect What's in Valkey

# List all keys for this session
keys = repo.client.keys("strands:session:user-42:*")
for k in sorted(keys):
    ttl = repo.client.ttl(k)
    print(f"{k}  (TTL: {ttl}s)")

# strands:session:user-42                          (TTL: 3598s)
# strands:session:user-42:agent:default            (TTL: 3597s)
# strands:session:user-42:agent:default:messages   (TTL: 3597s)
# strands:session:user-42:msg:<uuid>               (TTL: 3596s)
# strands:session:user-42:msg:<uuid>               (TTL: 3596s)

# Read all messages for the session
messages = repo.list_messages("user-42", "default")
for msg in messages:
    print(f"[{msg.role}] {str(msg.content)[:80]}")
